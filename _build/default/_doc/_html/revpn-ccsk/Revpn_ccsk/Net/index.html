Module Revpn_ccsk.Net

type place_id = string
type transition_id = string
type place = {
  1. p_id : place_id;
}
type transition = {
  1. t_id : transition_id;
  2. t_label : string;
}
type arc =
  1. | PT of place_id * transition_id * int
  2. | TP of transition_id * place_id * int

arc is the immediate relation between places and transitions. Here the type int plays the rol of the number of tokens.
